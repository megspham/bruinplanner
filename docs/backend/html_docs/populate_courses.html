114def encodeNumber(number):
115    """
116    Converts course API number to course human readable number (e.g "0152A M" -> "M152A")
117
118    Parameters
119    ----------
120    number : str
121        human readable number
122
123    Returns
124    -------
125    number : str
126        course API number
127    """
128    if number[0] == "M":
129        number = encodeNumber(number[1:]) + " M"
130    
131    numDigits = sum(c.isdigit() for c in number)
132    if numDigits <= 4:

196def parseCourse(department, number, start_term=None, type_=None):
197    """
198    Parses department name and number (and optionally start_term) of a certain course to obtain course info.
199    Course info is a tuple containing: course type, course name, department, course number, course start term, course units, course requisites, historical offerings

204        department of course in question
205    number: str

207    start_term: Optional[str]
208        start term of course in question
209    type_: str
210        category of requirement fulfilled by course in question
211
212    Returns
213    -------
214    courseInfo: tuple
215        tuple containing course type, course name, department, course number, course start term, course units, course requisites, historical offerings
216    """
217    if macros.VERBOSE:

219

221    if not start_term:
222        start_term = getCourseLatestStartTerm(department, number)
223        pass
224    
225    # get information about the class

227    class_requisites_info = makeAPIRequest(getCourseRequisitesURL(department, number, start_term))

229    # get units

235    else:
236        class_requisites = None
237
238    # get historical offerings- TODO
239    historical_offerings = None
240    
241    # create name
242    name = department + " " + decodeNumber(number)
243    
244    # construct and return courseInfo
245    courseInfo = (type_, name, department, number, start_term, units, class_requisites, historical_offerings)
246    
247    if macros.VERBOSE:
248        print(courseInfo)

251def parseCourses(courses):
252    """
253    Parses department name and number (and optionally start_term) of a list of courses to obtain course info for each.
254    courseInfos is a list of tuples, each containing: course type, course name, department, course number, course start term, course units, course requisites, historical offerings.
255    
256    Parameters
257    ----------
258    courses: List[(department, number, [start_term, [type_]])]
259        List of courses to be parsed using parseCourse
260
261    Returns
262    -------
263    courseInfos: List[(type_, name, department, number, start_term, units, class_requisites, historical_offerings)]
264        List of courseInfo tuples, each of which corresponds to a course in courses
265    """
266    courseInfos = []
267    for course in courses:
268        department = course[0]
269        number = course[1]
270        start_term = course[2] if len(course)>=3 else None
271        type_ = course[3] if len(course)>=4 else None
272        
273        courseInfo = parseCourse(department, number, start_term=start_term, type_=type_)
274        courseInfos.append(courseInfo)
275    return courseInfos
